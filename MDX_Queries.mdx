--------- Query 1 -------------------
SELECT   
   [Measures].[Sales Amount] ON 0,
    {[Dim Product].[Product Name].&[Contoso DVD Player M100 Black],
	[Dim Product].[Product Name].&[Contoso DVD Player M110 Silver],
	[Dim Product].[Product Name].&[Contoso DVD Player M120 White],
	[Dim Product].[Product Name].&[Contoso DVD Player M130 Grey],
	[Dim Product].[Product Name].&[Contoso DVD Player M140 Gold]
	} ON 1
FROM  
    [Contoso Retail DW]
WHERE 
	[Dim Date].[Calendar Year].&[2009]
